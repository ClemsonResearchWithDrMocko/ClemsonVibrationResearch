00036 /// Mappings and most-recently-seen values are stored in this list
00037 /** DBus messages are expected to come as two numbers; an integer then
00038    an integer or floating point number. The first is mapped onto a PID
00039    using this table. The second is the actual value to be passed.
00040 */
00041 struct dbus_simvals {
00042     int map_from; //< The first number in the message
00043     unsigned int pid; //< Is mapped to this OBDII PID
00044     struct obdservicecmd *pid_cmd; // Cache the cmd for this PID
00045     float most_recent; //< The most recent value we saw for this PID
00046     struct dbus_simvals *next; //< Yay linked lists
00047 };

00243     struct dbus_gen *g = gen;

00245     dbus_simgen_flushqueue(g, 1);

00247     if(0x00 == PID || 0x20 == PID || 0x40 == PID || 0x60 == PID) {

00250         else if(0x20 == PID) bits = g->supportedpids_20;
00251         else if(0x40 == PID) bits = g->supportedpids_40;
00252         else if(0x60 == PID) bits = g->supportedpids_60;

00255         *D = bits & 0xFF;
00256         bits >>= 8;
00257         *C = bits & 0xFF;
00258         bits >>= 8;
00259         *B = bits & 0xFF;
00260         bits >>= 8;
00261         *A = bits & 0xFF;

00263         return 4;

00279 DBusHandlerResult dbus_simgen_msgfilter(DBusConnection *connection,
00280         DBusMessage *message, void *gen) {
00281 
00282     // printf("In Message Filter\n");
00283 
00284     DBusMessageIter args;
00285     dbus_message_iter_init_append(message, &args);
00286

00288         fprintf(stderr, "Message has no arguments!\n"); 
00289 
00290     int key;  // We accept message tuples, an integer key
00291     unsigned int ukey; // Or unsigned [which we convert]
00292     double value_f; // And a floating point number (ints are converted)
00293 
00294     
00295     switch(dbus_message_iter_get_arg_type(&args)) {
00296         case DBUS_TYPE_INT32:
00297             dbus_message_iter_get_basic(&args, &key);
00298             break;
00299         case DBUS_TYPE_UINT32:
00300             dbus_message_iter_get_basic(&args, &ukey);
00301             key = (int)ukey;
00302             break;
00303         default:
00304             fprintf(stderr,"First argument isn't of type int\n");
00305             return DBUS_HANDLER_RESULT_NOT_YET_HANDLED;
00306     }
00307 
00308     if(dbus_message_iter_next(&args)) {
00309         switch(dbus_message_iter_get_arg_type(&args)) {
00310             case DBUS_TYPE_INT32: {
00311                 int value_i;
00312                 dbus_message_iter_get_basic(&args, &value_i);
00313                 value_f = value_i; // Set it in our floating point number
00314                 break;
00315             }
00316             case DBUS_TYPE_UINT32: {
00317                 unsigned int value_ui;
00318                 dbus_message_iter_get_basic(&args, &value_ui);
00319                 value_f = value_ui; // Set it in our floating point number
00320                 break;
00321             }
00322             case DBUS_TYPE_DOUBLE:
00323                 dbus_message_iter_get_basic(&args, &value_f);
00324                 break;
00325             default:
00326                 fprintf(stderr, "Second value in dbus message tuple must be number\n");
00327                 return DBUS_HANDLER_RESULT_NOT_YET_HANDLED;
00328                 break;
00329         }
00330     } else {
00331         fprintf(stderr, "Must have two values in dbus message\n");
00332         return DBUS_HANDLER_RESULT_NOT_YET_HANDLED;
00333     }
00334 
00335     // printf("DBus message contained everything we need: %i -> %f\n", key, value_f);
00336 
00337     struct dbus_simvals *v = dbus_simgen_findsimval_from(gen, key);
00338     if(NULL == v) {
00339         fprintf(stderr, "DBus Message tuple with an unconfigured key (%i)\n", key);
00340         return DBUS_HANDLER_RESULT_NOT_YET_HANDLED;
00341     }
00342 
00343     v->most_recent = value_f;
00344 
00345     return DBUS_HANDLER_RESULT_HANDLED;
