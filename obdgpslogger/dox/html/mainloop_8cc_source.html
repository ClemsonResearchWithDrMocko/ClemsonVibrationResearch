00490                 spaces?" ":"",
00491                 spaces?" ":"",
00492                 ecuaddress,

00498                 spaces?" ":"",
00499                 spaces?" ":"",
00500                 ecuaddress,
00501                 spaces?" ":"");
00502             break;
00503         case OBDHEADER_14230:
00504             ecuaddress = ecu->ecu_num + 0x10;

00506                 (unsigned)0x80 | messagelen, spaces?" ":"",
00507                 spaces?" ":"",
00508                 ecuaddress,
00509                 spaces?" ":"");
00510             break;
00511         case OBDHEADER_CAN29:

00518                 segments[0], spaces?" ":"",
00519                 segments[1], spaces?" ":"",
00520                 segments[2], spaces?" ":"",
00521                 segments[3], spaces?" ":"",
