00097 void *getobdcapabilities(int obd_serial_port, struct obdservicecmd **wishlist) {
00098     struct obdcapabilities *caps = (struct obdcapabilities *)malloc(sizeof(struct obdcapabilities));
00099     caps->next = NULL;
00100     caps->pid = 0x00;
00101     struct obdcapabilities *curr_cap = caps;
00102 
00103     unsigned int obdbytes[4];
00104     int bytes_returned;
00105     enum obd_serial_status cap_status;
00106     unsigned int current_cmd = 0x00;
00107 
00108     // getobdcapabilities_guess(obd_serial_port, wishlist, caps);
00109     // return caps;
00110 
00111     // Set if we should try to guess, instead;
00112     int must_guess = 0;
00113 
00114     while(1) {
00115 
00116         cap_status = getobdbytes(obd_serial_port, 0x01, current_cmd, 0,
00117             obdbytes, sizeof(obdbytes)/sizeof(obdbytes[0]), &bytes_returned, 1);
00118 
00119         if(OBD_SUCCESS != cap_status || 4 != bytes_returned) {
00120             fprintf(stderr, "Couldn't get obd bytes for cmd %02X\n", current_cmd);
00121             return caps;
00122         }
00123 
00124         unsigned long val;
00125         val = (unsigned long)obdbytes[0]*(256*256*256) +
00126             (unsigned long)obdbytes[1]*(256*256) +
00127             (unsigned long)obdbytes[2]*(256) +
00128             (unsigned long)obdbytes[3];
00129 
00130         if(0x00 == current_cmd && 0 == val) {
00131             // If this is 0x20, then the car must have provided *something*
00132             must_guess = 1;
00133             break;
00134         }
00135 
00136         int currbit;
00137         int c;
00138         for(c=current_cmd+1, currbit=31 ; currbit>=0 ; currbit--, c++) {
00139             // "cleaner" would be to set wishlist to be the whole obdcmd list and do the search
00140             //  But no point for the actual logic we need, here.
00141             int in_wishlist = 1;
00142             if(wishlist != NULL) {
00143                 in_wishlist = 0;
00144                 int i;
00145                 for(i=0;NULL != wishlist[i];i++) {
00146                     if(wishlist[i]->cmdid == c) {
00147                         in_wishlist = 1;
00148                         break;
00149                     }
00150                 }
00151             }
00152             if(val & ((unsigned long)1<<currbit)) {
00153                 if(in_wishlist) {
00154                     struct obdcapabilities *nextcap = (struct obdcapabilities *)malloc(sizeof(struct obdcapabilities));
00155                     nextcap->next = NULL;
00156                     nextcap->pid = c;
00157                     curr_cap->next = nextcap;
00158                     curr_cap = nextcap;
00159                 }
00160             }
00161         }
00162 
00163         if(obdbytes[3]&0x01) {
00164             current_cmd += 0x20;
00165         } else {
00166             break;
00167         }
00168     }
00169 
00170     if(must_guess) {
00171         fprintf(stderr, "Warning: Car reported no PIDs supported. Experimentally guessing instead\n");
00172         getobdcapabilities_guess(obd_serial_port, wishlist, caps);
00173     }
