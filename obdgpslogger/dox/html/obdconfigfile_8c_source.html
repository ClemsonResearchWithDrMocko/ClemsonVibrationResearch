00047 /// Get "a" valid home dir in which to store a dotfile

00049     static char homedir[MAX_PATH] = "\0";
00050 
00051     char *envhome;
00052     if(0 != strlen(homedir)) {
00053         return homedir;

00056         envhome = getenv("HOME");
00057         if(NULL != envhome && *envhome) {
00058             snprintf(homedir,sizeof(homedir), "%s/", envhome);
00059             return homedir;

00063         envhome = getenv("APPDATA");
00064         if(NULL != envhome && *envhome) {
00065             snprintf(homedir,sizeof(homedir), "%s", envhome);
00066             return homedir;

00296 int obd_configCmds(const char *log_columns, struct obdservicecmd ***cmds) {
00297     int cols = 0;
00298     *cmds = NULL;
00299     const char *toklist=",: "; // Seriously, we only want comma-separated though
00300 
00301     // Goind to do this twice. First time, get a count to allocate.
00302     char *cmdlist = strdup(log_columns);
00303     if(NULL == cmdlist) return -1;
00304 
00305     char *currcmd = strtok(cmdlist, toklist);
00306     while(currcmd) {
00307         struct obdservicecmd *c;
00308 
00309         unsigned int cmdpid;
00310         if(NULL != (c = obdGetCmdForColumn(currcmd))) {
00311             cols++;
00312         } else if(1 == sscanf(currcmd, "%2X", &cmdpid)) {
00313             if(NULL != (c = obdGetCmdForPID(cmdpid))) {
00314                 cols++;
00315             } else {
00316                 printf("Warning: Couldn't find column for PID '%s'. Possible config file problem\n", currcmd);
00317             }
00318         } else {
00319             printf("Warning: Couldn't find column '%s'. Possible config file problem\n", currcmd);
00320         }
00321         currcmd = strtok(NULL, toklist);
00322     }
00323     free((void *)cmdlist);
00324 
00325     // Second time, do the allocation then the assignement
00326     *cmds = (struct obdservicecmd **)malloc((cols+1) * sizeof(struct obdservicecmd *)); // 1 == NULL sentinel
00327 
00328     int currcol = 0;
00329     cmdlist = strdup(log_columns);
00330     if(NULL == cmdlist) return -1;
00331 
00332     currcmd = strtok(cmdlist, toklist);
00333     while(currcmd) {
00334         struct obdservicecmd *c;
00335 
00336         unsigned int cmdpid;
00337         if(NULL != (c = obdGetCmdForColumn(currcmd))) {
00338             (*cmds)[currcol++] = c;
00339         } else if(1 == sscanf(currcmd, "%2X", &cmdpid)) {
00340             if(NULL != (c = obdGetCmdForPID(cmdpid))) {
00341                 (*cmds)[currcol++] = c;
00342             }
00343         }
00344         currcmd = strtok(NULL, toklist);
00345     }
00346     free((void *)cmdlist);
00347 
00348     (*cmds)[currcol] = NULL;

00350     return cols;
