00056 int logger_simgen_create(void **gen, const char *seed) {

00058     if(NULL == filename || 0 == strlen(filename)) {
00059         fprintf(stderr, "Must use filename of log as the seed\n");
00060         return 1;
00061     }

00064     int rc;
00065     rc = sqlite3_open_v2(filename, &db, SQLITE_OPEN_READONLY, NULL);
00066     if( SQLITE_OK != rc ) {
00067         fprintf(stderr, "Can't open database %s: %s\n", filename, sqlite3_errmsg(db));
00068         sqlite3_close(db);
00069         return 1;
00070     }
00071 
00072     struct logger_gen *g = (struct logger_gen *)malloc(sizeof(struct logger_gen));
00073     if(NULL == g) {
00074         fprintf(stderr,"Couldn't allocate memory for logger generator\n");
00075         return 1;
00076     }
00077 
00078     g->db = db;
00079 
00080     // Get the supported PIDs according to the database
00081     g->supportedpids_00 = 0x01; // We can support getting higher PIDs without supporting anything else
00082     g->supportedpids_20 = 0x01;
00083     g->supportedpids_40 = 0x00;
00084     g->supportedpids_60 = 0x00;
00085 
00086     sqlite3_stmt *pragma_stmt; // The stmt for gathering table_info
00087     const char *dbend; // ignored handle for sqlite
00088     rc = sqlite3_prepare_v2(g->db, "PRAGMA table_info(obd)", -1, &pragma_stmt, &dbend);
00089     if(SQLITE_OK != rc) {
00090         printf("Couldn't get table info in database: %s\n", sqlite3_errmsg(g->db));
00091         sqlite3_finalize(pragma_stmt);
00092 
00093         sqlite3_close(db);
00094         free(g);
00095         return 1;
00096     }
00097 
00098     while(SQLITE_DONE != sqlite3_step(pragma_stmt)) {
00099         const char *columnname = sqlite3_column_text(pragma_stmt, 1);
00100         if(0 == strcmp(columnname, "time") || 0 == strcmp(columnname, "trip")) {
00101             continue;
00102         }
00103 
00104         struct obdservicecmd *cmd = obdGetCmdForColumn(columnname);
00105 
00106         if(NULL == cmd) {
00107             printf("Couldn't find cmd for column %s\n", columnname);
00108             continue;
00109         }
00110 
00111         unsigned int pid = cmd->cmdid;
00112 
00113         if(pid <= 0x20) {
00114             g->supportedpids_00 |= ((unsigned long)1<<(0x20 - pid));
00115         } else if(pid > 0x20 && pid <= 0x40) {
00116             g->supportedpids_20 |= ((unsigned long)1<<(0x40 - pid));
00117         } else if(pid > 0x40 && pid <= 0x60) {
00118             g->supportedpids_40 |= ((unsigned long)1<<(0x60 - pid));
00119         } else if(pid > 0x60 &&  pid <= 0x80) {
00120             g->supportedpids_60 |= ((unsigned long)1<<(0x80 - pid));
00121         } else {
00122             fprintf(stderr,"Don't support PIDs this high in sim yet: %i\n", pid);
00123         }
00124     }
00125 
00126     sqlite3_finalize(pragma_stmt);
00127     // Got the supported PIDs
00128 
00129 
00130     // Get the starting time of the database
00131     char time_select_sql[2048];
00132 
00133     // Taking our best guess means interpolating the value.
00134     snprintf(time_select_sql, sizeof(time_select_sql), "SELECT MIN(obd.time), MAX(obd.time) FROM obd");
00135 
00136     sqlite3_stmt *select_time_stmt; // Our actual select statement
00137     rc = sqlite3_prepare_v2(g->db, time_select_sql, -1, &select_time_stmt, &dbend);
00138 
00139     if(SQLITE_OK != rc) {
00140         printf("Couldn't prepare select %s: %s\n", time_select_sql, sqlite3_errmsg(g->db));
00141         sqlite3_finalize(select_time_stmt);
00142         return 1;
00143     }
00144 
00145 
00146     sqlite3_step(select_time_stmt); // We only step once - that's all we asked for.
00147 
00148     g->min_databasetime = sqlite3_column_double(select_time_stmt, 0);
00149     g->max_databasetime = sqlite3_column_double(select_time_stmt, 1);
00150 
00151     sqlite3_finalize(select_time_stmt);
00152     // Got the start time of the database
00153 
00154     if(0 != gettimeofday(&(g->simstart), NULL)) {
00155         fprintf(stderr, "Couldn't get time of day\n");
00156         sqlite3_close(db);
00157         free(g);
00158         return 1;
00159     }
00160 
00161     *gen = g;
00162     return 0;
00163 }

00171 int logger_simgen_getvalue(void *gen, unsigned int mode, unsigned int PID, unsigned int *A, unsigned int *B, unsigned int *C, unsigned int *D) {
00172     struct logger_gen *g = gen;
00173 
00174     if(0x00 == PID || 0x20 == PID || 0x40 == PID || 0x60 == PID) {
00175             unsigned long bits;
00176             if(0x00 == PID) bits = g->supportedpids_00;
00177             else if(0x20 == PID) bits = g->supportedpids_20;
00178             else if(0x40 == PID) bits = g->supportedpids_40;
00179             else if(0x60 == PID) bits = g->supportedpids_60;
00180             else return 0;
00181 
00182             *D = bits & 0xFF;
00183             bits >>= 8;
00184             *C = bits & 0xFF;
00185             bits >>= 8;
00186             *B = bits & 0xFF;
00187             bits >>= 8;
00188             *A = bits & 0xFF;
00189 
00190             return 4;
00191     }
00192 
00193     struct obdservicecmd *cmd = obdGetCmdForPID(PID);
00194     if(NULL == cmd || NULL == cmd->db_column || 0 == strlen(cmd->db_column)) {
00195             fprintf(stderr, "Requested unsupported PID\n");
00196             return 0;
00197     }
00198 
00199     // Getting here means we need to look up a real value.
00200     struct timeval currtime;
00201     if(0 != gettimeofday(&currtime, NULL)) {
00202             fprintf(stderr, "Couldn't get time of day\n");
00203             return 0;
00204     }
00205     struct timeval dt;
00206     dt.tv_sec = currtime.tv_sec - g->simstart.tv_sec;
00207     dt.tv_usec = currtime.tv_usec - g->simstart.tv_usec;
00208 
00209     // Time we're aiming for in the SELECT, taking into account when the database starts
00210     double seltime = g->min_databasetime + (double)dt.tv_sec+(double)dt.tv_usec/1000000.0f;
00211 
00212     while(seltime > g->max_databasetime) {
00213         seltime -= (g->max_databasetime - g->min_databasetime);
00214     }
00215 
00216     char sql[2048];
00217 
00218     // Taking our best guess means interpolating the value.
00219     snprintf(sql, sizeof(sql), "SELECT "
00220                 "(s.%s + ((%f-s.time)/(e.time-s.time))*(e.%s - s.%s)) AS est%s "
00221                 "FROM obd s, obd e "
00222                 "WHERE s.time = (SELECT MAX(obd.time) FROM obd WHERE time < %f) "
00223                 "AND     e.time = (SELECT MIN(obd.time) FROM obd WHERE time >= %f)",
00224                 cmd->db_column, seltime, cmd->db_column, cmd->db_column, cmd->db_column, seltime, seltime);
00225 
00226     // printf("SQL Select:\n%s\n", sql);
00227 
00228     sqlite3_stmt *select_stmt; // Our actual select statement
00229     const char *dbend; // ignored handle for sqlite
00230     int rc = sqlite3_prepare_v2(g->db, sql, -1, &select_stmt, &dbend);
00231     if(SQLITE_OK != rc) {
00232         // printf("Couldn't prepare select %s: %s\n", sql, sqlite3_errmsg(g->db));
00233         sqlite3_finalize(select_stmt);
00234         return 0;
00235     }
00236 
00237 
00238     sqlite3_step(select_stmt); // We only step once - that's all we asked for.
00239 
00240     double val = sqlite3_column_double(select_stmt, 0);
00241     int retval = cmd->convrev(val, A, B, C, D);
00242 
00243     sqlite3_finalize(select_stmt);
00244 
00245     return retval;
00246 }
