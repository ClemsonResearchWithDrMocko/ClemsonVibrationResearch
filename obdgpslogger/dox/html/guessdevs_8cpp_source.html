00030 int populateSerialDevs(Fl_Input_Choice *addto, const char *def) {
00031     glob_t names;
00032 
00033     addto->clear();
00034     if(NULL != def && 0 < strlen(def)) {
00035         addto->add(def);
00036         addto->value(0);
00037     }
00038 
00039     // Just take a bunch of semi-educated guesses:
00040 
00041     // This is the OSX standard way of doing things
00042     glob("/dev/cu.*", 0, NULL, &names);
00043     // Linux bluetooth
00044     glob("/dev/rfcomm*", GLOB_APPEND, NULL, &names);
00045     // Linux USB
00046     glob("/dev/ttyUSB*", GLOB_APPEND, NULL, &names);
00047     // Good ol' fashioned serial port
00048     glob("/dev/ttyS*", GLOB_APPEND, NULL, &names);
00049 
00050     size_t i;
00051     for(i=0; i<names.gl_pathc; i++) {
00052         addto->add(names.gl_pathv[i]);
00053     }
00054     
00055     globfree(&names);
00056 
00057     return 0;
00058 }
